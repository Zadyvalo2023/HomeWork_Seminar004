using System;
namespace PowerExample
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Введите число A: ");
            double a = Convert.ToDouble(Console.ReadLine());
            Console.Write("Введите степень B: ");
            int b = Convert.ToInt32(Console.ReadLine());
            double result = 1;
            for (int i = 0; i < b; i++)
            {
                result *= a;
            }
            Console.WriteLine($"{a} в степени {b} = {result}");
            Console.ReadKey();
        }
    }
}
В данном коде сначала мы запрашиваем у пользователя числа A и B через консольный ввод. Затем мы инициализируем переменную result значением 1 и запускаем цикл for, который проходит от 0 до B-1, в каждой итерации умножая result на A. Таким образом, на выходе мы получаем A в степени B, которую выводим в консоль.

=========================================================

using System; - это директива using, которая позволяет использовать функционал из пространства имен System;
namespace PowerExample - это определение пространства имен PowerExample, в котором содержится класс Program;
class Program - это определение класса Program, который содержит точку входа в программу - метод Main;
static void Main(string[] args) - это метод Main, который запускается при запуске программы. Он принимает массив строк в качестве аргументов и имеет модификаторы static и void, что означает, что он является статическим методом, который не возвращает никакого значения;
Console.Write("Введите число A: "); - это выводит в консоль текст "Введите число A: ", который запрашивает у пользователя ввод значения a;
double a = Convert.ToDouble(Console.ReadLine()); - это строка считывает значение, введенное пользователем, из консоли и преобразует его в тип double, а затем присваивает a;
Console.Write("Введите степень B: "); - это выводит в консоль текст "Введите степень B: ", который запрашивает у пользователя ввод значения b;
int b = Convert.ToInt32(Console.ReadLine()); - это строка считывает значение, введенное пользователем, из консоли и преобразует его в тип int, а затем присваивает b;
double result = 1; - это определяет переменную result и присваивает ей начальное значение 1;
for (int i = 0; i < b; i++) - это цикл for, который выполняется b раз. i - это счетчик цикла, который увеличивается на 1 с каждой итерацией;
result *= a; - это команда умножения значения result на a и присваивает результат обратно в переменную result;
Console.WriteLine($"{a} в степени {b} = {result}"); - это выводит в консоль результат возведения числа a в степень b, используя переменную result;
Console.ReadKey(); - это ожидание ввода пользователя, прежде чем закрыть консольное окно.
